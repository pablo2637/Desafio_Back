const { pool } = require('../configs/configPostgreSQL');

const {
    queriesRecycle
} = require('./queries');



/**
 * Inserta en la base de datos un reciclaje nuevo
 * @method modelCreateRecycle
 * @async
 * @param {String} user_id ID del usuario
 * @param {String} place_id ID del establecimiento
 * @param {Number} qty Cantidad a reciclar
 * @returns {json} 
 * @throws {Error}
 */
const modelCreateRecycle = async ({ user_id, place_id, qty }) => {

    let client, result;
    try {

        client = await pool.connect();

        const data = await client.query(queriesRecycle.createRecycle, [user_id, place_id, qty])

        data.rowCount != 0 ? result = data.rows : result = false;

    } catch (e) {
        throw e;

    } finally {
        client.release();

    };

    return result;
};
